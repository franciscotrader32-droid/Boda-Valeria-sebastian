    <script>
        lucide.createIcons();
        const overlay = document.getElementById('welcome-overlay');
        const content = document.getElementById('content');
        const audio = document.getElementById('musica-fondo');
        const btnAbrir = document.getElementById('btn-abrir');

        btnAbrir.addEventListener('click', () => {
            audio.play().catch(e => console.log("Audio block bypass"));
            overlay.classList.add('fade-out');
            setTimeout(() => {
                overlay.style.display = 'none';
                content.classList.remove('invisible');
                document.getElementById('audio-control').classList.remove('hidden');
                initCountdown();
                initReveal();
                initQR();
            }, 1000);
        });

        function initCountdown() {
            const target = new Date("2026-03-08T17:30:00-06:00").getTime();
            setInterval(() => {
                const now = new Date().getTime();
                const diff = target - now;
                if (diff < 0) return;
                document.getElementById('days').innerText = String(Math.floor(diff / (1000 * 60 * 60 * 24))).padStart(2, '0');
                document.getElementById('hours').innerText = String(Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60))).padStart(2, '0');
                document.getElementById('mins').innerText = String(Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60))).padStart(2, '0');
                document.getElementById('secs').innerText = String(Math.floor((diff % (1000 * 60)) / 1000)).padStart(2, '0');
            }, 1000);
        }

        function initReveal() {
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) entry.target.classList.add('active');
                });
            }, { threshold: 0.1 });
            document.querySelectorAll('.reveal').forEach(el => observer.observe(el));
        }

        function initQR() {
            new QRCode(document.getElementById("qrcode"), {
                text: "INV-VAL-SEB-LAURA-SILVA-2026",
                width: 120, height: 120,
                colorDark : "#2D2D2D", colorLight : "#ffffff"
            });
        }

        function enviarWhatsApp() {
            const pases = document.getElementById('num-pases').value;
            const alergias = document.getElementById('alergias').value || "Ninguna";
            const texto = `¡Hola! Soy Laura Silva. Confirmo mi asistencia.%0A🎫 Pases: ${pases}%0A🍽️ Alergias: ${alergias}`;
            window.open(`https://wa.me/527204519733?text=${texto}`, '_blank');
        }
